    def sortedSquares(self, nums: List[int]) -> List[int]:
        left = 0
        right = len(nums) - 1
        res = [0] * len(nums)
        count = len(nums) - 1

        while count >= 0:
            if abs(nums[left]) >= abs(nums[right]):
                res[count] = nums[left] * nums[left]
                left = left + 1
            else:
                res[count] = nums[right] * nums[right]
                right = right - 1
            count = count - 1

        return res
